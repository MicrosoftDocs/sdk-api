# CD3D11_BOX structure


## -description


Represents a box and provides convenience methods for creating boxes.

struct CD3D11_BOX : public D3D11_BOX
{
    CD3D11_BOX()
    {}
    explicit CD3D11_BOX( const D3D11_BOX&amp; o ) :
        D3D11_BOX( o )
    {}
    explicit CD3D11_BOX(

        back = Back;

};
